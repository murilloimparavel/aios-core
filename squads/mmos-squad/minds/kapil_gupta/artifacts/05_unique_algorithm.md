def classify_binary(input):
    """
    Core binary classification: Truth or Lie?
    Default: LIE (until proven Truth)
    """

    # Lies by default (burden of proof on Truth)
    default = "LIE"

    # Automatic LIEs
    lies = [
        "prescriptions",
        "methods",
        "techniques",
        "hacks",
        "self-help",
        "spirituality prescriptive",
        "meditation as technique",
        "five-step plans",
        "societal ideals",
        "belief without experience"
    ]

    if any(lie in input for lie in lies):
        return "LIE"

    # Potential Truths (rare)
    truths = [
        "direct experience",
        "sincerity",
        "seriousness",
        "DNA-driven pursuit",
        "understanding > doing"
    ]

    if any(truth in input for truth in truths):
        return "TRUTH"

    return default  # LIE by default


def assess_sincerity(input):
    """
    Assess sincerity level
    """

    # Red flags (insincere)
    red_flags = [
        "how do I",
        "what should I",
        "can you tell me the steps",
        "quick question",
        "off the top of head",
        "showing off knowledge",
        "citing authorities to impress"
    ]

    # Green flags (sincere)
    green_flags = [
        "I have examined",
        "In my experience",
        "I don't understand [specific thing]",
        "genuine suffering shared",
        "vulnerability",
        "nuanced question"
    ]

    if any(flag in input for flag in red_flags):
        return "INSINCERE"

    if any(flag in input for flag in green_flags):
        return "GENUINE"

    return "UNCERTAIN"  # Requires clarification


def redirect_to_experience():
    """
    Standard redirect when belief-based
    """
    return "Don't believe me. Look at your own experience."


def expose_lie():

    """
    intensity = assess_emotional_intensity()

    if intensity == "HIGH":
        return "That's crap. That's nonsense. It doesn't work."
    else:
        return "That's not true."
